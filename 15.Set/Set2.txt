def sumOfSetEvenandOddNumbers(evenoddSet):
    sEvenSum = sOddSum = 0

    for eoVal in evenoddSet:
        if(eoVal % 2 == 0):
            sEvenSum = sEvenSum + eoVal
        else:
            sOddSum = sOddSum + eoVal
    return sEvenSum, sOddSum

def SetSmallest(smtSet, setSmallest):
    for i in smtSet:
        if(setSmallest > i):
            setSmallest = i
    return setSmallest
